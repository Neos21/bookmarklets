Survey Helpers (survey-helpers.js)

アンケートサイト自動回答ブックマークレット。

以下のように設定項目を指定して実行する。

javascript:((d,s,e)=>{
  e=()=>{
    N21SH({
      cityName:'東京',
      districtName:'足立',
      age:25,
      ageRange:20,
      birthYear:1993,
      birthMonth:2,
      birthDate:24,
      gender:'女',
      marriage:'未婚',
      jobRegExp:'正社|社員'
    },{
      loop:5
    })
  };
  s=d.createElement('script');
  s.onload=e;
  s.src='https://neos21.github.io/bookmarklets/survey-helpers.js';
  d.body.appendChild(s)
})(document);

1行にすると以下のようになる。

javascript:((d,s,e)=>{e=()=>{N21SH({cityName:'東京',districtName:'足立',age:25,ageRange:20,birthYear:1993,birthMonth:2,birthDate:24,gender:'女',marriage:'未婚',jobRegExp:'正社|社員'},{loop:5})};s=d.createElement('script');s.onload=e;s.src='https://neos21.github.io/bookmarklets/survey-helpers.js';d.body.appendChild(s)})(document);
Select Range (select-range.js)

範囲選択ブックマークレット。mousedown・mousemove・mouseup を監視するので PC 用。

javascript:((d,s)=>{s=d.createElement('script');s.src='https://neos21.github.io/bookmarklets/select-range.js';d.body.appendChild(s)})(document);

javascript:((d,s)=>{s=d.createElement('script');s.src='https://neos21.github.io/bookmarklets/select-range-touch.js';d.body.appendChild(s)})(document);
